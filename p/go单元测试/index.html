package main

import (
    "fmt"
    "testing"
)


func TestIntMinTableDriven(t *testing.T) {
    var tests = []struct {
        a, b int
        want int
    }{
        {0, 1, 0},
        {1, 0, 0},
        {2, -2, -2},
        {0, -1, -1},
        {-1, 0, -1},
    }

    // t.Run å¯ä»¥è¿è¡Œä¸€ä¸ª â€œsubtestsâ€ å­æµ‹è¯•ï¼Œä¸€ä¸ªå­æµ‹è¯•å¯¹åº”è¡¨ä¸­ä¸€è¡Œæ•°æ®ã€‚ è¿è¡Œ go test -v æ—¶ï¼Œä»–ä»¬ä¼šåˆ†å¼€æ˜¾ç¤ºã€‚
    for _, tt := range tests {

        testname := fmt.Sprintf("%d,%d", tt.a, tt.b)
        t.Run(testname, func(t *testing.T) {
            ans := IntMin(tt.a, tt.b)
            if ans != tt.want {
                t.Errorf("got %d, want %d", ans, tt.want)
            }
        })
    }
}
