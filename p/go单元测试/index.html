package main

import (
    "fmt"
    "testing"

func TestIntMinTableDriven(t *testing.T) {
    var tests = []struct {
        a, b int
        want int
    }{
        {0, 1, 0},
        {1, 0, 0},
        {2, -2, -2},
        {0, -1, -1},
        {-1, 0, -1},
    }

    // t.Run ÂèØ‰ª•ËøêË°å‰∏Ä‰∏™ ‚Äúsubtests‚Äù Â≠êÊµãËØïÔºå‰∏Ä‰∏™Â≠êÊµãËØïÂØπÂ∫îË°®‰∏≠‰∏ÄË°åÊï∞ÊçÆ„ÄÇ ËøêË°å go test -v Êó∂Ôºå‰ªñ‰ª¨‰ºöÂàÜÂºÄÊòæÁ§∫„ÄÇ

        testname := fmt.Sprintf("%d,%d", tt.a, tt.b)
        t.Run(testname, func(t *testing.T) {
            ans := IntMin(tt.a, tt.b)
            if ans != tt.want {
                t.Errorf("got %d, want %d", ans, tt.want)
            }
        })
    }
}
