func (t *Todo) Add() (todoID int, err error) {

	// ËøûÊé•Êï∞ÊçÆÂ∫ì

	// ÊâßË°åÊ∑ªÂä†sql

	if err != nil {
		return

	_ = tx.Commit()

	todoID = int(lastTodoID)
	return
}
