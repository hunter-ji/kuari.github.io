const uploadRequest = (file: File, url: string) => {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest()
    xhr.open('PUT', url, true)
    xhr.send(file)
    xhr.onload = () => {
      if (xhr.status === 200)
        resolve(xhr.response)
      else
        reject(xhr.status)
    }
  })
}
